    <!-- Tailwind CSS CDN -->
    <script src="https://cdn.tailwindcss.com"></script>
    <!-- Google Fonts -->

        .status-btn.active-absent { background-color: #ef4444; color: white; border-color: #ef4444; }

    <div class="max-w-2xl mx-auto">
        <!-- 헤더 -->
        <header class="mb-8 text-center">
            <h1 class="text-3xl font-bold text-slate-800">📋 학생 출석 체크</h1>
            <p class="text-slate-500 mt-2">실시간으로 출석 데이터를 관리하세요.</p>
        </header>

        <!-- 입력 섹션 -->
        <div class="bg-white p-6 rounded-2xl shadow-sm border border-slate-200 mb-8">
            <div class="space-y-4">
                <div>
                    <label class="block text-sm font-semibold text-slate-700 mb-1">학생 이름</label>
                    <input type="text" id="studentName" placeholder="이름을 입력하세요" 
                        class="w-full px-4 py-3 rounded-lg border border-slate-300 focus:ring-2 focus:ring-blue-500 focus:outline-none transition-all">
                </div>
                
                <div>
                    <label class="block text-sm font-semibold text-slate-700 mb-2">상태 선택</label>
                    <div class="grid grid-cols-3 gap-3">
                        <button onclick="setStatus('출석')" id="btn-present" class="status-btn py-3 px-4 rounded-lg border border-slate-200 font-medium hover:bg-slate-50 transition-all text-slate-600">출석</button>
                        <button onclick="setStatus('결석')" id="btn-absent" class="status-btn py-3 px-4 rounded-lg border border-slate-200 font-medium hover:bg-slate-50 transition-all text-slate-600">결석</button>
                        <button onclick="setStatus('지각')" id="btn-late" class="status-btn py-3 px-4 rounded-lg border border-slate-200 font-medium hover:bg-slate-50 transition-all text-slate-600">지각</button>
                    </div>
                </div>

                <button onclick="submitAttendance()" id="submitBtn"
                    class="w-full bg-slate-800 text-white font-bold py-4 rounded-xl hover:bg-slate-700 transition-all shadow-lg active:scale-[0.98]">
                    기록하기
                </button>
            </div>
        </div>

        <!-- 리스트 섹션 -->
        <div class="space-y-4">
            <div class="flex justify-between items-center">
                <h2 class="text-xl font-bold text-slate-800">최근 출석 현황</h2>
                <span id="totalCount" class="text-sm bg-slate-200 px-3 py-1 rounded-full font-medium text-slate-600">총 0건</span>
            </div>
            
            <div id="attendanceList" class="space-y-3">
                <!-- 데이터가 여기에 실시간으로 추가됨 -->
                <div class="text-center py-10 text-slate-400">데이터를 불러오는 중...</div>
            </div>
        </div>
    </div>

    <!-- Firebase SDK (Modular) -->
    <script type="module">
        import { initializeApp } from "https://www.gstatic.com/firebasejs/10.7.1/firebase-app.js";
        import { getFirestore, collection, addDoc, serverTimestamp, query, orderBy, onSnapshot, deleteDoc, doc } 
        from "https://www.gstatic.com/firebasejs/10.7.1/firebase-firestore.js";

        // 🔥 본인의 Firebase 설정값으로 교체하세요!
        const firebaseConfig = {
            apiKey: "YOUR_API_KEY",
            authDomain: "YOUR_PROJECT_ID.firebaseapp.com",
            projectId: "YOUR_PROJECT_ID",
            storageBucket: "YOUR_PROJECT_ID.appspot.com",
            messagingSenderId: "YOUR_SENDER_ID",
            appId: "YOUR_APP_ID"
        };

        // Firebase 초기화
        const app = initializeApp(firebaseConfig);
        const db = getFirestore(app);
        const attendanceCol = collection(db, "attendance");

        // 상태 관리 변수
        let selectedStatus = '';

        // 상태 선택 함수 (Global 등록)
        window.setStatus = (status) => {
            selectedStatus = status;
            // 버튼 스타일 초기화
            document.querySelectorAll('.status-btn').forEach(btn => {
                btn.classList.remove('active-present', 'active-absent', 'active-late');
            });
            // 선택된 버튼 하이라이트
            if(status === '출석') document.getElementById('btn-present').classList.add('active-present');
            if(status === '결석') document.getElementById('btn-absent').classList.add('active-absent');
            if(status === '지각') document.getElementById('btn-late').classList.add('active-late');
        };

        // 데이터 저장 함수
        window.submitAttendance = async () => {
            const nameInput = document.getElementById('studentName');
            const name = nameInput.value.trim();

            if (!name) return alert('이름을 입력해 주세요.');
            if (!selectedStatus) return alert('상태를 선택해 주세요.');

            try {
                await addDoc(attendanceCol, {
                    name: name,
                    status: selectedStatus,
                    timestamp: serverTimestamp()
                });
                
                // 입력창 초기화
                nameInput.value = '';
                selectedStatus = '';
                document.querySelectorAll('.status-btn').forEach(btn => {
                    btn.classList.remove('active-present', 'active-absent', 'active-late');
                });
            } catch (e) {
                console.error("Error adding document: ", e);
                alert('데이터 저장 중 오류가 발생했습니다.');
            }
        };

        // 데이터 삭제 함수
        window.deleteEntry = async (id) => {
            if(confirm('이 기록을 삭제하시겠습니까?')) {
                await deleteDoc(doc(db, "attendance", id));
            }
        };

        // 실시간 데이터 리스너
        const q = query(attendanceCol, orderBy("timestamp", "desc"));
        onSnapshot(q, (querySnapshot) => {
            const listContainer = document.getElementById('attendanceList');
            document.getElementById('totalCount').innerText = `총 ${querySnapshot.size}건`;
            
            if (querySnapshot.empty) {
                listContainer.innerHTML = '<div class="text-center py-10 text-slate-400 bg-white rounded-xl border border-dashed">기록이 없습니다.</div>';
                return;
            }

            listContainer.innerHTML = '';
            querySnapshot.forEach((doc) => {
                const data = doc.data();
                const date = data.timestamp ? data.timestamp.toDate().toLocaleString('ko-KR', { hour12: true, month: 'short', day: 'numeric', hour: '2-digit', minute: '2-digit' }) : '방금 전';
                
                let statusColor = '';
                if(data.status === '출석') statusColor = 'bg-green-100 text-green-700';
                else if(data.status === '결석') statusColor = 'bg-red-100 text-red-700';
                else statusColor = 'bg-amber-100 text-amber-700';

                const html = `
                    <div class="flex items-center justify-between bg-white p-4 rounded-xl shadow-sm border border-slate-100 group animate-in fade-in duration-300">
                        <div class="flex items-center gap-4">
                            <div class="w-10 h-10 bg-slate-100 rounded-full flex items-center justify-center font-bold text-slate-500">
                                ${data.name.charAt(0)}
                            </div>
                            <div>
                                <h3 class="font-bold text-slate-800">${data.name}</h3>
                                <p class="text-xs text-slate-400">${date}</p>
                            </div>
                        </div>
                        <div class="flex items-center gap-3">
                            <span class="px-3 py-1 rounded-full text-xs font-bold ${statusColor}">
                                ${data.status}
                            </span>
                            <button onclick="deleteEntry('${doc.id}')" class="text-slate-300 hover:text-red-500 transition-colors p-1">
                                <svg xmlns="http://www.w3.org/2000/svg" class="h-5 w-5" fill="none" viewBox="0 0 24 24" stroke="currentColor">
                                    <path stroke-linecap="round" stroke-linejoin="round" stroke-width="2" d="M19 7l-.867 12.142A2 2 0 0116.138 21H7.862a2 2 0 01-1.995-1.858L5 7m5 4v6m4-6v6m1-10V4a1 1 0 00-1-1h-4a1 1 0 00-1 1v3M4 7h16" />
                                </svg>
                            </button>
                        </div>
                    </div>
                `;
                listContainer.innerHTML += html;
            });
        });
    </script>
